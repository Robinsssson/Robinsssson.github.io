LU2 = @(t)(t-tau(2))./(tau(3)-tau(2));
fx = @(t)L0(t).*X0+L1(t).*X1+L2(t).*X2;
fu = @(t)LU1(t).*U1+LU2(t).*U2;

figure;
subplot(2, 1, 1);
fplot(fx, [-1, 1]);
subplot(2, 1, 2);
fplot(fu, [-1, 1]);

function  f = myfun(x)
U1 = x(4);
U2 = x(5);

end

function [c, ceq] = mycon(x, D)
% x = {X0, X1, X2, U1, U2}
% c <= 0
% ceq = 0

X2 = x(3);

ceq(3) = D(1, 1) * X0 + D(1, 2) * X1 + D(1, 3) * X2 + X1^2 - U1;
ceq(4) = D(2, 1) * X0 + D(2, 2) * X1 + D(2, 3) * X2 + X2^2 - U2;
end
